    32	func (f *Facade) Query(query string, args ...any) ([]map[string]any, error) {
    33		if f == nil || f.DB == nil {
    34			return nil, fmt.Errorf("%s is not configured", f.name())
    35		}
    36		if f.Readonly && !isReadQuery(query) {
    37			return nil, fmt.Errorf("%s only allows SELECT/WITH queries", f.name())
    38		}
    39		rows, err := f.DB.QueryContext(context.Background(), query, args...)
    40		if err != nil {
    41			return nil, err
    42		}
    43		defer rows.Close()
    44	
    45		cols, err := rows.Columns()
    46		if err != nil {
    47			return nil, err
    48		}
    49		ret := []map[string]any{}
    50		for rows.Next() {
    51			vals := make([]any, len(cols))
    52			scan := make([]any, len(cols))
    53			for i := range vals {
    54				scan[i] = &vals[i]
    55			}
    56			if err := rows.Scan(scan...); err != nil {
    57				return nil, err
    58			}
    59			row := make(map[string]any, len(cols))
    60			for i, col := range cols {
    61				row[col] = normalizeSQLiteValue(vals[i])
    62			}
    63			ret = append(ret, row)
    64		}
    65		if err := rows.Err(); err != nil {
    66			return nil, err
    67		}
    68		return ret, nil
    69	}
